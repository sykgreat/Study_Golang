windows下编译：
    windows:
        直接go build -o main.exe main.go进行编译
    linux:
        配置交叉编译环境：
            SET CGO_ENABLED=0
            SET GOOS=linux
            SET GOARCH=amd64
            在执行go build -o main main.go进行编译
        修改回原来的环境：
            SET CGO_ENABLED=1
            SET GOOS=windows
            SET GOARCH=amd64
    mac:
        配置交叉编译环境：
            SET CGO_ENABLED=0
            SET GOOS=darwin
            SET GOARCH=amd64
            在执行go build -o main main.go进行编译
        修改回原来的环境：
            SET CGO_ENABLED=1
            SET GOOS=windows
            SET GOARCH=amd64
